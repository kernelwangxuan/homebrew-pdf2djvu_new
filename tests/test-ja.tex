\input common

\pdfpagewidth 6pt
\pdfpageheight 13pt

\immediate\pdfobj{
<<
    /CIDSystemInfo <<
        /Ordering (Japan1)
        /Registry (Adobe)
    >>
    /Subtype /CIDFontType0
    /Type /Font
>>
}

\immediate\pdfobj{
<<
    /BaseFont /System
    /DescendantFonts [ \the\pdflastobj\space 0 R ]
    /Encoding /Identity-H
    /Subtype /Type0
    /Type /Font
>>
}

\edef\addpdfresources{
    \pdfpageresources{
        /Font << /F99 \the\pdflastobj \space 0 R >>
    }
}
\addpdfresources

\pdfliteral{
    BT
    /F99 12 Tf
    1 0 0 1 0 -10 Tm [<0002>]TJ
    ET
}
